    // v1.3 Phase 22 Plan 02 — Stream slug + URL derivation (per L-22 + HUB-OUT-06).
    //

    // The on-disk YAML key is what go2rtc serves; the URL the user copies must match.

    export type OutputType = 'loxone-mjpeg' | 'frigate-rtsp';

    ```ts
    // v1.3 Phase 22 Plan 02 — Derived hub_state machine (per L-18, A1 in RESEARCH).
    //
    // Pure read of three signals — no DB writes, no side effects:
    //   - settings.protect_hub_enabled
    //   - protect_hub_bridges.status
    //   - hub_onboarding_state pointer (presence + step + error)
    //
    // The 5-state enum L-18: disabled | starting | enabled | stopping | error.
    // Toggle UI gates the toggle-disabled state on { starting, stopping }.
    import { getSetting } from '$lib/server/services/settings';
    import { getBridgeStatus } from '$lib/server/orchestration/protect-hub/bridge-lifecycle';
    import { getPointer } from '$lib/server/orchestration/protect-hub/wizard-state';

    export type HubState = 'disabled' | 'starting' | 'enabled' | 'stopping' | 'error';

    export function getHubState(): HubState {
        const enabled = getSetting('protect_hub_enabled') === 'true';
        const bridge = getBridgeStatus();
        const pointer = getPointer();

        // error has highest priority — surface the bad state regardless of others
        if (bridge?.status === 'failed') return 'error';
        if (pointer?.error) return 'error';

        // starting: provisioning OR wizard in-progress mid-flight
        if (bridge?.status === 'provisioning') return 'starting';
        if (pointer && pointer.status === 'in_progress' && pointer.step >= 1 && pointer.step <= 5) {
            return 'starting';
        }

        // enabled: feature flag on AND bridge running
        if (enabled && bridge?.status === 'running') return 'enabled';

        // stopping: feature flag was on, bridge no longer running, no active pointer
        // (full lifecycle for stopping is owned by P23; in P22 we just surface the in-between)
        if (enabled && bridge && bridge.status !== 'running') return 'stopping';

        return 'disabled';
    }
    ```
